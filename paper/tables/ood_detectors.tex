    \toprule
    \textbf{Detector} & \textbf{Formula} & \textbf{Description} \\
    \midrule
    \textbf{Entropy}~\cite{dsd_baseline} 
      & $H(x) = - \sum_{y} p_\theta(y|x) \log p_\theta(y|x)$ 
      & Predictive entropy of the model output distribution. \\[0.75em]
         
    \textbf{GradNorm}~\cite{gradients_distshift} 
      & $||\nabla_{\theta}\, H(x)||_{2}$ 
      & $l_2$ norm of the gradient of entropy w.r.t.\ model parameters. \\[0.75em]
         
    \textbf{Energy}~\cite{dsd_energy} 
      & $E(x) = -\log \sum_{y} e^{f_\theta(x)_y}$ 
      & Energy score, i.e.\ negative log-sum-exp of logits $f_\theta(x)$. \\[0.75em]
         
    \textbf{kNN}~\cite{knn_dsd} 
      & $\min_{x_i \in \mathcal{D}_{\text{train}}} ||z(x) - z(x_i)||_2$ 
      & Distance between test encoding $z(x)$ and its nearest neighbour $z_k(x)$ in the training set. The original paper extends this to multiple $k$ values. \\[0.75em]
         
    \textbf{MSP}~\cite{dsd_baseline} 
      & $\max(p_\theta(y|x))$ 
      & Maximum softmax score. \\[0.75em]
         
    \textbf{Typicality}~\cite{typicality_dsd} 
      & $-\log p_{\text{GLoW}}(x)$ 
      & Negative log-likelihood under the GLoW generative model. \\[0.75em]
    \bottomrule
\end{tabularx}
